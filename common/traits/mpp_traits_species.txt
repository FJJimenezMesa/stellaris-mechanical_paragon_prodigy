# ================================
# Variables base para traits de especies MPP
# ================================
@base_ai_weight = 0              		# Peso base AI (IA le asigna al rasgo al generar o seleccionar especies automáticamente.)
@default_ai_weight = @base_ai_weight  	# Peso AI por defecto
@default_crossbreding = 1.0      		# Probabilidad de que esta especie pueda cruzarse con otras especies compatibles (1.0 = 100%)

# ================================
# 🧬 Species Trait: trait_mpp_parangon_legacy
# "Legado Parangon"
# ================================
# Descripción: Rasgo poderoso que otorga longevidad, estabilidad y bonificaciones de producción e investigación.
# Aplicación: Especies de tipo BIOLOGICAL y LITHOID, con posibilidad de cruzamiento avanzado.
# Enfoque: Rareza, longevidad, resiliencia y sinergias productivas.
# ================================
mpp_trait_parangon_legacy = {
    cost = -8
	icon = "gfx/interface/icons/traits/trait_mpp_parangon_legacy.dds"
	
	ai_weight = { @default_ai_weight }

	allowed_archetypes = { BIOLOGICAL LITHOID }
	species_potential_add = {
		can_do_advanced_gene_modding = yes
	}
	potential_crossbreeding_chance = @default_crossbreding

    modifier = {
		leader_age = 50

    	mpp_species_leader_bonus = yes
		mpp_bio_population_bonus = yes
		mpp_job_production_bonus = yes
		mpp_planet_research_production_bonus = yes
    }
}

# ================================
# 🧬 Species Trait: trait_parangon_creation
# "Creación Parangon"
# ================================
# Descripción: Potente rasgo mecánico que incrementa producción, felicidad, eficiencia de líderes y trabajos, 
# además de mejorar ensamblaje y definir costes de esclavos.
# Aplicación: Exclusivo para especies ROBOT y MACHINE.
# Enfoque: Optimización mecánica, colonización eficiente y supremacía tecnológica.
# ================================
mpp_trait_parangon_creation = {
    cost = 10
    icon = "gfx/interface/icons/traits/trait_parangon_creation.dds"

	ai_weight = { @default_ai_weight }

    allowed_archetypes = { ROBOT MACHINE }

    modifier = {
		mpp_species_leader_bonus = yes
		mpp_robotic_population_bonus = yes
		mpp_job_production_bonus = yes
		mpp_research_production_bonus = yes
    }

    slave_cost = {
        energy = 750
    }

    assembly_score = {
        modifier = {
            add = 1.5
            from = { has_mining_designation = yes }
        }
        modifier = {
            add = 0.5
            from = { has_rural_designation = yes }
        }
    }
}
